# YouTube Downloader (Fully Working — Independent) This version actually downloads videos using backend (yt-dlp). Frontend-only downloaders cannot work due to YouTube restrictions. --- # Folder Structure ``` youtube-downloader/ │ ├── server.js ├── package.json ├── index.html └── README.md ``` --- # Step 1 — package.json ```json { "name": "youtube-downloader", "version": "1.0.0", "main": "server.js", "scripts": { "start": "node server.js" }, "dependencies": { "express": "^4.18.2", "cors": "^2.8.5", "yt-dlp-exec": "^1.0.2" } } ``` --- # Step 2 — server.js ```javascript const express = require("express"); const cors = require("cors"); const ytdlp = require("yt-dlp-exec"); const app = express(); app.use(cors()); app.get("/download", async (req, res) => { const url = req.query.url; const format = req.query.format || "mp4"; if (!url) return res.send("Missing URL"); try { res.header("Content-Disposition", "attachment"); ytdlp(url, { output: "-", format: format === "mp3" ? "bestaudio" : "bestvideo+bestaudio", }).stdout.pipe(res); } catch (e) { res.send("Error downloading video"); } }); app.listen(3000, () => { console.log("Server running on http://localhost:3000"); }); ``` --- # Step 3 — index.html ```html YouTube Downloader
